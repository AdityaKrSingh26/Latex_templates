\begin{frame}{Problem \& Motivation}
    \begin{columns}[T]
        \column{0.55\textwidth}
        The global RTTVP market is valued at $\approx$USD~3.2 billion. Leading platforms process over \textbf{8~billion data points daily} across 1.5~billion annual shipments.

        \vspace{0.4em}
        A critical data quality failure: a single shipment tracked by \textbf{multiple vehicles simultaneously}, producing position histories no single physical object could have generated.

        \vspace{0.4em}
        \textbf{Data quality dimensions at risk:}
        \begin{itemize}
            \item \textbf{Accuracy:} GPS coordinates vs.\ actual location
            \item \textbf{Completeness:} gaps from tunnels, battery loss
            \item \textbf{Consistency:} physically plausible trajectories
            \item \textbf{Validity:} correct formats and value ranges
        \end{itemize}

        \column{0.42\textwidth}
        \begin{alertblock}{Business Impact}
            \begin{itemize}
                \item Customer confusion \& erratic tracking pages
                \item Incorrect ETAs \& dock-door scheduling
                \item Billing disputes (phantom mileage)
                \item Operational blindness
                \item SLA violations \& expedited freight costs
            \end{itemize}
        \end{alertblock}

        \vspace{0.3em}
        {\footnotesize Poor data quality costs organisations an average of \$12.9M annually.}
    \end{columns}
\end{frame}

% ---- Slide 8: Graph Theory Background ----
\begin{frame}{Graph Model \& Formulation}
    \vspace*{-0.5em}
    \begin{columns}[T]
        \column{0.38\textwidth}
        \begin{block}{Notation}
            \footnotesize
            \begin{tabular}{ll}
                $p_i = (\text{lat}, \text{lon}, t)$ & Position \\
                $P = \{p_1, \ldots, p_n\}$ & Sorted sequence \\
                $V_{\max} = 250$\,km/h & Max feasible speed \\
                $\tau = 1800$\,s & Jitter window \\
                $T_H = 28800$\,s & Handover window \\
                $\alpha = 0.10$ & Min cluster \% \\
                $s_{\min} = 5$ & Min cluster size \\
            \end{tabular}
        \end{block}

        \vspace{0.2em}
        \begin{block}{Reachability Edge}
            \footnotesize
            Edge $(p_i, p_j)$ exists iff:
            \[
            \underbrace{d(p_i,p_j)}_{\text{Haversine}} \leq V_{\max} \cdot (t_j - t_i)
            \]
            Nodes = GPS positions.\quad Graph is always a \textbf{DAG} (time-ordered, no cycles).
        \end{block}

        \column{0.58\textwidth}
        \begin{block}{Jitter Tolerance}
            \footnotesize
            GPS noise handled by checking \textbf{multiple recent} positions in window $\tau$:
            $\exists\, p_i \in W(p_j, C, \tau) : \mathrm{reachable}(p_i, p_j)$\\
            $W$: positions in cluster $C$ within $\tau = 30$\,min before $p_j$.
        \end{block}

        \vspace{0.2em}
        \begin{block}{Handover Detection}
            \footnotesize
            Legitimate inter-vehicle transfers detected via \textbf{boundary positions}:\\
            $\exists\, p_i \in \mathrm{last}_k(C_1),\; p_j \in \mathrm{first}_k(C_2) :$
            $|t_i - t_j| \leq T_H \;\wedge\; \mathrm{reachable}(p_i, p_j)$\\
            $k = 5$ boundary positions, $T_H = 8$\,h.
        \end{block}

        \vspace{0.2em}
        \begin{block}{Significance Filter}
            \footnotesize
            A cluster is \textbf{significant} iff $|C| \geq \max(s_{\min},\; \alpha \cdot n)$.\\
            Filters GPS jitter / outlier clusters from the multi-vehicle decision.
        \end{block}
    \end{columns}
\end{frame}

% ---- Slide 10: Key Theorems & Correctness ----
\begin{frame}{Key Theorems \& Correctness}
    \footnotesize
    \begin{columns}[T]
        \column{0.50\textwidth}
        \begin{block}{Theorem 1: Single Component $\Rightarrow$ Single Vehicle}
            \footnotesize One weakly connected component $\Rightarrow$ all positions consistent with one vehicle.\\
            {\scriptsize\textit{Proof:} Every position reachable from every other $\Rightarrow$ feasible single-vehicle trajectory.}
        \end{block}

        \vspace{0.15em}
        \begin{alertblock}{Theorem 2: Multiple Large Components $\Rightarrow$ Multiple Vehicles}
            \footnotesize $G$ has $\geq 2$ components each with $\geq 10\%$ of positions $\Rightarrow$ multiple vehicles.\\
            {\scriptsize\textit{Proof:} Two large components cannot both be noise; no edges between them $\Rightarrow$ separate vehicles.}
        \end{alertblock}

        \vspace{0.15em}
        \begin{block}{Theorem 3: Handover Preserves Validity}
            \footnotesize Two components merging via valid handover $\Rightarrow$ valid single-vehicle tracking with transfer.
        \end{block}

        \column{0.47\textwidth}
        \begin{block}{Lemma 1: Completeness}
            \footnotesize All positions from one vehicle $\Rightarrow$ algorithm returns \textbf{TRUE}.\\
            {\scriptsize\textit{Proof:} Single vehicle positions are mutually reachable; clustering yields one component $\Rightarrow$ TRUE.}
        \end{block}

        \vspace{0.15em}
        \begin{block}{Lemma 2: Soundness}
            \footnotesize Two vehicles at distant locations $\Rightarrow$ algorithm returns \textbf{FALSE}.\\
            {\scriptsize\textit{Proof:} Impossible speeds prevent edges across vehicles $\Rightarrow$ separate significant clusters $\Rightarrow$ FALSE.}
        \end{block}

        \vspace{0.15em}
        \begin{block}{Mapping to Standard Problems}
            \scriptsize
            \begin{tabular}{ll}
                \toprule
                \textbf{Our Problem} & \textbf{Standard Problem} \\
                \midrule
                Initial clustering & Connected components \\
                Cluster merging & Iterative pairwise union \\
                Significance test & Component size threshold \\
                Handover detection & Boundary reachability \\
                \bottomrule
            \end{tabular}
        \end{block}
    \end{columns}
\end{frame}


\begin{frame}{Algorithm: SingleVehicleCheck}
    \vspace*{-0.8em}
    \begin{columns}[T]
        \column{0.50\textwidth}
        \begin{block}{Key Sub-Algorithms}
            \scriptsize
            \begin{enumerate}
                \item \textsc{PreprocessPositions}: filter \& sort
                \item \textsc{BuildInitialClusters}: online clustering
                \item \textsc{IsReachable}: Haversine + speed check
                \item \textsc{GetRecentPositions}: jitter window
                \item \textsc{MergeClustersViaHandovers}: iterative merge
                \item \textsc{CanHandover}: boundary reachability
                \item \textsc{MakeDecision}: significance threshold
            \end{enumerate}
        \end{block}

        \vspace{0.15em}
        \begin{block}{Clustering (Phase 2): Pseudocode}
            \scriptsize
            \textbf{for each} position $p$ chronologically:\\
            \quad \textbf{for each} existing cluster $C$ (newest first):\\
            \quad\quad $\mathit{recent} \leftarrow$ last 3 positions of $C$ within $\tau$\\
            \quad\quad \textbf{if} any $q \in \mathit{recent}$ reachable from $p$: assign $p$ to $C$; \textbf{break}\\
            \quad \textbf{if} not assigned: create new cluster $\{p\}$\\
            Merge clusters via handover predicate (iterate until stable).
        \end{block}

        \vspace{0.15em}
        \begin{block}{Decision (Phase 3)}
            \scriptsize
            $\mathit{threshold} \leftarrow \max(s_{\min},\; \alpha \cdot n)$\quad
            0 sig.\ clusters $\to$ \textbf{UNCERTAIN} $\cdot$
            1 $\to$ \textbf{TRUE} $\cdot$
            $\geq\!2$ $\to$ \alert{\textbf{FALSE}}
        \end{block}

        \column{0.46\textwidth}
        \centering
        \begin{tikzpicture}[
            phase/.style={rectangle, draw=DarkBlue, fill=blue!8, text width=4.0cm, minimum height=0.6cm, align=center, rounded corners=4pt, font=\footnotesize\bfseries},
            arrow/.style={-{Stealth[length=2.5mm]}, thick, DarkBlue}
        ]
            \node[phase] (p1) {Phase 1: Preprocessing \\ {\scriptsize Filter nulls/NaN, validate ranges, sort}};
            \node[phase, below=0.4cm of p1] (p2) {Phase 2: Clustering \\ {\scriptsize Online reachability + handover merging}};
            \node[phase, below=0.4cm of p2] (p3) {Phase 3: Decision \\ {\scriptsize Count significant clusters $\to$ result}};
            \draw[arrow] (p1) -- (p2);
            \draw[arrow] (p2) -- (p3);
        \end{tikzpicture}

        \vspace{0.4em}
        \begin{block}{Haversine Distance}
            \scriptsize
            \[
            d = 2R \cdot \arcsin\!\sqrt{\sin^2\!\tfrac{\Delta\phi}{2} + \cos\phi_i\cos\phi_j\sin^2\!\tfrac{\Delta\lambda}{2}}
            \]
            $R = 6{,}371$\,km.\quad Edge exists iff $d \leq V_{\max}\cdot\Delta t$,\; $V_{\max}=250$\,km/h.
        \end{block}
    \end{columns}
\end{frame}
